// دالة التنقل بين الصفحات function showPage(id) { // إخفاء كل الصفحات document.querySelectorAll('.page').forEach(p => p.classList.remove('active')); // إلغاء تفعيل كل الأزرار document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active')); // إظهار الصفحة المطلوبة document.getElementById('page-' + id).classList.add('active'); // تفعيل الزر الصحيح في القائمة const idx = ['home','about','games','hub','facts','gallery'].indexOf(id); if (idx !== -1) { document.querySelectorAll('.nav-btn')[idx].classList.add('active'); }
