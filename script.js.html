    { icon:'🔴', name:'البلاستيد الأخضر', desc:'يوجد في الخلايا النباتية. يحتوي الكلوروفيل ويُجري التمثيل الضوئي.' },
    { icon:'🟣', name:'جهاز غولجي', desc:'يُعدّل ويُغلّف البروتينات ويُرسلها إلى وجهتها داخل أو خارج الخلية.' },
    { icon:'🧬', name:'الكروموسومات', desc:'تتكوّن من DNA ملفوف حول بروتينات الهيستون. الإنسان يمتلك 46 كروموسوماً.' },
    { icon:'🌀', name:'الحلزون المزدوج DNA', desc:'جزيء الـ DNA على شكل حلزون مزدوج مكوّن من خيطين متوازيين ومتكاملين.' },
    { icon:'⭕', name:'الفجوة المركزية', desc:'توجد في الخلايا النباتية وتحتوي على العصارة الخلوية وتدعم شكل الخلية.' },
    ];
    let points = 0;
    let badges = [];
    let currentGame = null;
    let currentQ = 0;
    let answered = false;
    let score = 0;
    function showPage(id) {
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
    document.getElementById('page-' + id).classList.add('active');
    const idx = ['home','about','games','hub','facts','gallery'].indexOf(id);
    document.querySelectorAll('.nav-btn')[idx].classList.add('active');
    }
    function renderGames() {
    const g = document.getElementById('games-grid');
    g.innerHTML = GAMES.map(game => `
    <div class="game-card">
    <div class="game-header">
    <div class="game-icon">${game.icon}</div>
    <div class="game-meta">
    <div class="game-title">${game.title}</div>
    <div class="game-sub">${game.sub}</div>
    </div>
    </div>
    <div class="game-topics">${game.questions.slice(0,2).map(q=>q.q).join('<br>')}<br><span style="color:var(--accent)">+ ${game.questions.length - 2} أسئلة أخرى</span></div>
    <button class="play-btn" onclick="startGame(${game.id})">▶ ابدأي اللعبة</button>
    </div>
    `).join('');
    }
    function renderHub() {
    const h = document.getElementById('hub-topics');
    h.innerHTML = TOPICS.map((t,i) => `
    <div class="topic-card">
    <div class="topic-header" onclick="toggleTopic(${i})">
    <span style="font-size:24px">${t.icon}</span>
    <div class="topic-title">${t.title}</div>
    <span class="topic-toggle" id="tog-${i}">▼</span>
    </div>
    <div class="topic-body" id="topic-body-${i}">
    <p class="topic-text">${t.text}</p>
    <ul class="topic-points">${t.points.map(p=>`<li>${p}</li>`).join('')}</ul>
    <button class="quiz-mini-btn" onclick="startGame(${t.gameId}); showPage('games')">🎮 اختبري نفسك باللعبة!</button>
    </div>
    </div>
    `).join('');
    }
    function toggleTopic(i) {
    const body = document.getElementById('topic-body-' + i);
    const tog = document.getElementById('tog-' + i);
    const open = body.classList.toggle('open');
    tog.textContent = open ? '▲' : '▼';
    }

    <div class="gallery-img">${g.icon}</div>
    <div class="gallery-label">
    <div class="gallery-name">${g.name}</div>

    </div>
    </div>

    }
    function startGame(id) {
    currentGame = GAMES[id];
    currentQ = 0; score = 0; answered = false;
    document.getElementById('quiz-overlay').classList.add('open');
    showQuestion();
    }
    function showQuestion() {
    const q = currentGame.questions[currentQ];
    const total = currentGame.questions.length;
    const pct = (currentQ / total * 100).toFixed(0);
    document.getElementById('quiz-content').innerHTML = `
    <div class="quiz-title">${currentGame.icon} ${currentGame.title}</div>
    <div class="quiz-progress">سؤال ${currentQ + 1} من ${total}</div>
    <div class="progress-bar"><div class="progress-fill" style="width:${pct}%"></div></div>
    <div class="quiz-q">${q.q}</div>
    <div class="quiz-opts">
    ${q.opts.map((o,i) => `<button class="opt-btn" id="opt-${i}" onclick="answer(${i})">${o}</button>`).join('')}
    </div>
    <div class="quiz-feedback" id="qfb"></div>
    <button class="quiz-next-btn" id="qnext" onclick="nextQ()">${currentQ + 1 < total ? 'التالي ←' : 'النتيجة 🏁'}</button>
    `;
    }
    function answer(i) {
    if (answered) return;
    answered = true;
    const q = currentGame.questions[currentQ];
    const opts = document.querySelectorAll('.opt-btn');
    opts.forEach(b => b.disabled = true);
    const fb = document.getElementById('qfb');
    if (i === q.ans) {
    opts[i].classList.add('correct');
    fb.textContent = '✅ إجابة صحيحة!'; fb.style.color = '#2db87a';
    score++;
    } else {
    opts[i].classList.add('wrong');
    opts[q.ans].classList.add('correct');
    fb.textContent = '❌ إجابة خاطئة. الصحيح: ' + q.opts[q.ans]; fb.style.color = '#e24b4a';
    }
    document.getElementById('qnext').style.display = 'block';
    }
    function nextQ() {
    currentQ++;
    answered = false;
    if (currentQ < currentGame.questions.length) {
    showQuestion();
    } else {
    showResult();
    }
    }
    function showResult() {
    const total = currentGame.questions.length;
    const pct = Math.round(score / total * 100);
    let badge = '', msg = '', stars = 0;
    if (pct === 100) { badge = '🏆'; msg = 'ممتاز! أداء مثالي!'; stars = 3; }
    else if (pct >= 70) { badge = currentGame.badge; msg = 'رائع! أداء جيد جداً!'; stars = 2; }
    else if (pct >= 50) { badge = '⭐'; msg = 'جيد! استمري في التعلم!'; stars = 1; }
    else { badge = '💪'; msg = 'حاولي مرة أخرى!'; stars = 0; }
    points += score * 10;
    if (badge && !badges.includes(badge)) badges.push(badge);
    updateBar();
    document.getElementById('quiz-content').innerHTML = `
    <div class="quiz-score-box">
    <div class="quiz-badge">${badge}</div>
    <div class="quiz-score-big">${score}/${total}</div>
    <div class="stars">${[1,2,3].map(s=>`<span class="star" style="opacity:${s<=stars?1:0.25}">⭐</span>`).join('')}</div>
    <div class="quiz-score-msg">${msg}<br><span style="font-size:13px;color:var(--muted)">حصلتِ على ${score*10} نقطة!</span></div>
    <div style="display:flex;gap:10px;justify-content:center;margin-top:1rem;">
    <button onclick="startGame(${currentGame.id})" style="background:var(--primary);color:#fff;border:none;padding:10px 20px;border-radius:10px;font-family:inherit;font-size:13px;font-weight:700;cursor:pointer;">🔄 أعيدي اللعب</button>
    <button onclick="closeQuiz()" style="background:#eee;color:var(--text);border:none;padding:10px 20px;border-radius:10px;font-family:inherit;font-size:13px;font-weight:700;cursor:pointer;">✕ إغلاق</button>
    </div>
    </div>
    `;
    }
    function updateBar() {
    document.getElementById('total-pts').textContent = points;
    const bl = document.getElementById('badge-list');
    bl.innerHTML = badges.length ? badges.map(b=>`<span class="badge-item">${b}</span>`).join('') : '<span style="opacity:0.5;font-size:12px">لا يوجد بعد</span>';
    }
    function closeQuiz() {
    document.getElementById('quiz-overlay').classList.remove('open');
    }
    renderGames();
    renderHub();
    renderFacts();
    renderGallery();
